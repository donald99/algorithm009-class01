     * Maintains the heap property while deleting an element.
     */
    private void heapifyDown(int i) {
        int child;
        int temp = heap[i];                // ä¿å­˜è¦æ›¿æ¢çš„å€¼
        while (kthChild(i, 1) < heapSize) {// å­èŠ‚ç‚¹ç´¢å¼•æ˜¯å¦å°äºå †å¤§å°

            if (temp >= heap[child]) {     // ä¿å­˜è¦æ›¿æ¢çš„å€¼å¤§äºå„¿å­èŠ‚ç‚¹ä¸­æœ€å¤§çš„ï¼Œåœæ­¢ä¸‹æµ®æ“ä½œ

    }

    private int maxChild(int i) {          // æ‰¾åˆ°å½“å‰èŠ‚ç‚¹çš„å·¦å³å­èŠ‚ç‚¹ä¸­æœ€å¤§å€¼çš„ç´¢å¼•
        int leftChild = kthChild(i, 1);
        int rightChild = kthChild(i, 2);
        return heap[leftChild] > heap[rightChild] ? leftChild : rightChild;
    }
